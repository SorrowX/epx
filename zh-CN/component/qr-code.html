QRCode 二维码 ​

把文本转成二维码

基础用法 ​

使用text属性来生成二维码内容


vue
<template>
  <el-qr-code :text="text" />
  <br />
  <el-input v-model="text" style="width: 240px" />
</template>

<script lang="ts" setup>
import { ref } from 'vue'
const text = ref<string>('Summer')
</script>

图标 ​

使用icon-src地址在二维码中间生成一个图标

使用icon-config来配置icon的圆角 内边距 大小 背景色

vue
<template>
  <el-space :size="32">
    <el-qr-code
      :text="text"
      icon-src="https://avatars.githubusercontent.com/u/20411966?v=4&size=64"
    />

    <el-qr-code
      :text="text"
      icon-src="https://avatars.githubusercontent.com/u/20411966?v=4&size=64"
      :icon-config="{
        radius: 4,
        padding: 0,
        size: 30,
        backgroundColor: '#fff',
      }"
    />
  </el-space>
</template>

<script lang="ts" setup>
import { ref } from 'vue'
const text = ref<string>('Summer')
</script>

尺寸 ​

使用size来指定尺寸


vue
<template>
  <el-button-group>
    <el-button @click="add">尺寸+10</el-button>
    <el-button @click="subtract">尺寸-10</el-button>
  </el-button-group>
  <br />
  <el-qr-code :text="text" :size="size" />
</template>

<script lang="ts" setup>
import { ref } from 'vue'
const text = ref<string>('Summer')

const size = ref<number>(150)

const add = () => (size.value += 10)

const subtract = () => (size.value -= 10)
</script>

纠错 ​

使用QR Code options来指定选项

使用选项 errorCorrectionLevel 属性即可，options选项中的值优先级最大

vue
<template>
  <div style="margin-bottom: 16px">
    <el-radio-group v-model="radio">
      <el-radio-button label="low" value="L" />
      <el-radio-button label="medium" value="M" />
      <el-radio-button label="quartile" value="Q" />
      <el-radio-button label="high" value="H" />
    </el-radio-group>
  </div>
  <el-qr-code :text="text" :options="options" />
</template>

<script lang="ts" setup>
import { computed, ref } from 'vue'
const text = ref<string>('https://cn.vuejs.org/')
const radio = ref('L')

const options = computed(() => {
  return {
    errorCorrectionLevel: radio.value,
  }
})
</script>

下载 ​

使用组件实例的downloadQRCode方法即可

vue
<template>
  <el-space :size="32">
    <el-flex vertical>
      <el-qr-code
        ref="qrCodeRef"
        :text="text"
        :margin="1"
        icon-src="https://avatars.githubusercontent.com/u/20411966?v=4&size=64"
      />
      <el-button @click="handleClick">下载</el-button>
    </el-flex>

    <el-flex vertical>
      <el-qr-code ref="qrCodeRef1" :margin="1" :text="text" />
      <el-button @click="handleClick1">下载</el-button>
    </el-flex>
  </el-space>
</template>

<script lang="ts" setup>
import { ref } from 'vue'

const text = ref<string>('Summer')

const qrCodeRef = ref()
const qrCodeRef1 = ref()

const handleClick = () => {
  qrCodeRef.value.downloadQRCode()
}
const handleClick1 = () => {
  qrCodeRef1.value.downloadQRCode('code.png')
}
</script>

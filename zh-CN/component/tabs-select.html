TabsSelect 标签页选择器 ​

多页签选择器，支持分组、列表、树形等结构

使用v-model来绑定选中的值。

    :tabs="tabs"

    @change="handleChange"

    v-model="value"
    :tabs="tabs"
    style="width: 320px"
    :loading="loading"
    :remote-method="remoteMethod"
    @change="handleChange"
  />
</template>

<script setup lang="ts">
import { ref } from 'vue'

          label: `${query}1`,
          value: `${query}1`,
        },
        {
          label: `${query}2`,
          value: `${query}2`,

  }
